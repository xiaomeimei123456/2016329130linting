	num2 = parseFloat(document.myform.txtNum2.value);	
	switch(obj){
		case "+":
			result=num1 + num2;
			break;
		case "-":
			result=num1 - num2;
			break;
		case "*":
			result=num1 * num2;
			break;
		case "/":
			if(num2!=0)
				result=num1 / num2;
			else
				result="除数不能为0，请重新输入！";	
							
			break;			
		}
		document.myform.txtResult.value=result;	
	}
